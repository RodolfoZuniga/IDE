# Inicio del Programa
DECLARE NUM int
DECLARE V int
DECLARE WAL int
WRITE "Ingrese el valor de NUM: "
READ NUM
WRITE "Ingrese el valor de V: "
READ V
t0 = NUM == 1
IF_FALSE t0 GOTO L0
t1 = 100 - V
WAL = t1
GOTO L1
L0:
t2 = NUM == 2
IF_FALSE t2 GOTO L2
t3 = 100 + V
WAL = t3
GOTO L3
L2:
t4 = NUM == 3
IF_FALSE t4 GOTO L4
t5 = 100 / V
WAL = t5
GOTO L5
L4:
WAL = 0
L5:
L3:
L1:
WRITE "El resultado de WAL es: "
WRITE WAL
# Fin del Programa
HALT
