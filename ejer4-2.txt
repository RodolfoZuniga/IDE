main {
    int NUM;
    int V;
    int WAL;

    cout << "Ingrese el valor de NUM: ";
    cin >> NUM;
    cout << "Ingrese el valor de V: ";
    cin >> V;

    if NUM == 1 then
        WAL = 100 * V;
    else
        if NUM == 2 then
            WAL = 100 ** V;
        else
            if NUM == 3 then
                WAL = 100 / V;
            else
                WAL = 0;
            end
        end
    end

    cout << "El resultado de WAL es: ";
    cout << WAL;
}
